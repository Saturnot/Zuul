002/**
003 * Classe Command - une commande du jeu d'aventure Zuul.
004 *
005 * @author ROUSTANT Jolan
006 */
007public class Command
008{
009    /// ATTRIBUTS ///
010    private String aCommandWord;
011    private String aSecondWord;
012    
013    /// CONSTRUCTEUR ///
014    public Command(final String pCommandWord, final String pSecondWord)
015    {
016        this.aCommandWord = pCommandWord;
017        this.aSecondWord = pSecondWord;
018    }
019    
020    /// ACCESSEURS ///
021    public String getCommandWord(){return this.aCommandWord;}
022    public String getSecondWord(){return this.aSecondWord;}
023    
024    /// FCT hasSecondWord ///
025    public boolean hasSecondWord()
026    {
027        return(this.getSecondWord() != null);
028    }
029    
030    /// FCT isUnknown ///
031    public boolean isUnknown()
032    {
033        return(this.getCommandWord() == null);
034    }
035} // Command
