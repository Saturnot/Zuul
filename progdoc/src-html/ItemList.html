001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Décrivez votre classe ItemList ici.
006 *
007 * @author (votre nom)
008 * @version (un numéro de version ou une date)
009 */
010public class ItemList
011{
012    // variables d'instance - remplacez l'exemple qui suit par le vôtre
013    private HashMap <String, Item> aItems;
014    private String aLocation;
015
016    /**
017     * Constructeur d'objets de classe ItemList
018     */
019    public ItemList(final String pLocation)
020    {
021        this.aItems = new HashMap <String, Item>();
022        this.aLocation = pLocation;
023    }
024
025    public Item getItem(final String pItem)
026    {
027        return this.aItems.get(pItem);
028    }
029    
030    public HashMap <String, Item> getItems()
031    {
032        return this.aItems;
033    }
034    
035    public String getItemsDescription()
036    {
037        if(this.aItems.isEmpty())
038        {
039            return "Il n'y a pas d'objet dans " + this.aLocation;
040        }
041        String vString = "Objets dans " + this.aLocation + " :";
042        Set<String> vKeys = this.aItems.keySet();
043        for(String vS : vKeys)
044        {
045            vString += "\n - " + vS;
046        }
047        return vString;
048    }
049    
050    public void addItem(final String pName, final Item pItem){
051        this.aItems.put(pName, pItem);
052    }
053    
054    public void supItem (final String pName){
055        this.aItems.remove(pName);
056    }
057    
058    public int getItemListWeight()
059    {
060        if(this.aItems.isEmpty())
061        {
062            return 0;
063        }
064        int vWeight = 0;
065        Set<String> vKeys = this.aItems.keySet();
066        for(String vS : vKeys)
067        {
068            vWeight += this.getItem(vS).getWeight();
069        }
070        return vWeight;
071    }
072    
073}
