001import java.util.HashMap;
002import java.util.Set;
003/**
004 * Classe Room - un lieu du jeu d'aventure Zuul.
005 *
006 * @author ROUSTANT Jolan
007 */
008public class Room
009{   
010    /// ATRIBUTS ///
011    private String aDescription;
012    private HashMap<String, Room> aExits;
013    private HashMap<String,Item> aItemList;
014    private String aImageName;
015    
016    /// CONSTRUCTEUR ///
017    /**
018     * Constructor of the Room class
019     * @param string description of the room
020    */
021    public Room(final String pDescriptionLieu, final String pImage)
022    {
023        this.aDescription = pDescriptionLieu;
024        this.aExits = new HashMap<String, Room>();
025        this.aItemList = new HashMap<String, Item>();
026        this.aImageName = pImage;
027    }//Room
028    
029    /// ACCESSEUR ///
030    /**
031     * getter description
032     * @return description of a room
033     */
034    public String getDescription(){ return this.aDescription; }//getDescription
035    
036    /// MODIFICATEUR ///
037    /**
038     * seter
039     * @param String of a direction and Room of the neighbor room
040     */
041    public void setExits(final String pDirection,final Room pNeighbor)
042    {
043        this.aExits.put(pDirection, pNeighbor);
044    }//setExits
045    
046    public void addItem(final String pName, final Item pItem)
047    {
048        this.aItemList.put(pName, pItem);
049    }
050    
051    public Item getItem(final String pItemName)
052    {
053        return this.aItemList.get(pItemName);
054    }
055    
056    public void supItem(final String pItemName)
057    {
058        this.aItemList.remove(pItemName);
059    }
060    
061    /**
062     * getter exit
063     * @param String of a direction
064     * @return direction of a exit
065     */
066    public Room getExit(String pDirection)
067    {
068        return this.aExits.get(pDirection);
069    }//getExit
070    
071    /**
072     * getter exit Sting
073     * @return String list exits
074     */
075    public String getExitString()
076    {
077        String vExitsList = "Les sorties visibles ici sont : ";
078        Set<String> vKeys = this.aExits.keySet();
079        for(String exit : vKeys)
080        {
081            vExitsList += " " + exit; 
082        }//for
083        return vExitsList;
084    }//getExitString
085    
086    public String getItemDescription()
087    {
088        String vItemDescription = "Les objets sont : \n";
089        Set<String> vKeys = this.aItemList.keySet();
090        for(String vItem : vKeys)
091        {
092            vItemDescription += " " + vItem + "\n"; 
093        }//for
094        return vItemDescription;
095    }
096    
097    /**
098     * method that 
099     * @return a String with the description of the room and the exits
100     */
101    public String getLongDescription()
102    {
103        return this.aDescription + "\n" + this.getExitString() + "\n" + this.getItemDescription();
104    }//getLongDescription
105    
106    /**
107     * 
108     */
109    public String getImageName()
110    {
111        return this.aImageName;
112    }
113} // Room
